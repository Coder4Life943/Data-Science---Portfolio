# Select relevant features
features = [
    'Age', 'Gender', 'Smoking', 'Alcohol Consumption',
    'BMI', 'Cholesterol Level', 'Triglyceride Level', 'Blood Pressure','Diabetes','Low HDL Cholesterol',
]
target = 'Heart Disease Status'

# Drop rows with missing values in selected features
data_model = data[features + [target]].dropna()

# Encode categorical columns
categorical_cols = ['Gender', 'Smoking', 'Alcohol Consumption','Diabetes','Low HDL Cholesterol']
data_encoded = pd.get_dummies(data_model, columns=categorical_cols, drop_first=True)

# Encode target variable
label_encoder = LabelEncoder()
data_encoded[target] = label_encoder.fit_transform(data_encoded[target])  # Yes = 1, No = 0

# Split into features and target
X = data_encoded.drop(target, axis=1)
y = data_encoded[target]
